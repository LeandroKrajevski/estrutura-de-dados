#include <stdio.h>


int ordenar(int *a, int *b, int *c) {
    int temp;


    if(*a > *b){ temp=*a; *a=*b; *b=temp; }
    if(*a > *c){ temp=*a; *a=*c; *c=temp; }
    if(*b > *c){ temp=*b; *b=*c; *c=temp; }


    if(*a == *b && *b == *c)
        return 1;
    else
        return 0;
}


int main() {
    int x, y, z;


    scanf("%d %d %d", &x, &y, &z);


    int iguais = ordenar(&x, &y, &z);


    printf("Ordenados: %d %d %d\n", x, y, z);
    printf("Todos iguais? %d\n", iguais);


    return 0;
}

#include <stdio.h>
#include <stdlib.h>


// Nó da pilha
typedef struct No {
    char dado;
    struct No *prox;
} No;


// Pilha
typedef struct {
    No *topo;
} Pilha;


// Inicializar pilha
void init(Pilha *p) {
    p->topo = NULL;
}


// Verificar se está vazia
int isEmpty(Pilha *p) {
    return p->topo == NULL;
}


// Push
void push(Pilha *p, char c) {
    No novo = (No) malloc(sizeof(No));
    novo->dado = c;
    novo->prox = p->topo;
    p->topo = novo;
}


// Pop
char pop(Pilha *p) {
    if (isEmpty(p)) return '\0';


    No *temp = p->topo;
    char valor = temp->dado;
    p->topo = temp->prox;
    free(temp);


    return valor;
}


// Verifica correspondência
int combina(char abre, char fecha) {
    return (abre == '(' && fecha == ')') ||
           (abre == '{' && fecha == '}') ||
           (abre == '[' && fecha == ']');
}


int main() {
    Pilha p;
    init(&p);


    char str[100];
    printf("Digite a expressao: ");
    scanf("%s", str);


    for (int i = 0; str[i] != '\0'; i++) {
        char c = str[i];


        if (c == '(' || c == '{' || c == '[') {
            push(&p, c);
        }
        else if (c == ')' || c == '}' || c == ']') {
            if (isEmpty(&p)) {
                printf("Expressao INVALIDA\n");
                return 0;
            }


            char topo = pop(&p);


            if (!combina(topo, c)) {
                printf("Expressao INVALIDA\n");
                return 0;
            }
        }
    }


    if (isEmpty(&p))
        printf("Expressao VALIDA\n");
    else
        printf("Expressao INVALIDA\n");


    return 0;
}

#include <stdio.h>
#include <stdlib.h>


// Struct cliente
typedef struct Cliente {
    int id;
    int tempoAtendimento;
    struct Cliente *prox;
} Cliente;


// Fila
typedef struct {
    Cliente *inicio;
    Cliente *fim;
} Fila;


// Inicializar
void init(Fila *f) {
    f->inicio = NULL;
    f->fim = NULL;
}


// Enqueue
void enqueue(Fila *f, int id, int tempo) {
    Cliente novo = (Cliente) malloc(sizeof(Cliente));
    novo->id = id;
    novo->tempoAtendimento = tempo;
    novo->prox = NULL;


    if (f->fim == NULL) {
        f->inicio = f->fim = novo;
    } else {
        f->fim->prox = novo;
        f->fim = novo;
    }
}


// Dequeue
Cliente* dequeue(Fila *f) {
    if (f->inicio == NULL) return NULL;


    Cliente *temp = f->inicio;
    f->inicio = temp->prox;


    if (f->inicio == NULL)
        f->fim = NULL;


    return temp;
}


int main() {
    Fila f;
    init(&f);


    int n;
    printf("Quantidade de clientes: ");
    scanf("%d", &n);


    for (int i = 0; i < n; i++) {
        int id, tempo;
        printf("ID e tempo de atendimento: ");
        scanf("%d %d", &id, &tempo);
        enqueue(&f, id, tempo);
    }


    int tempoEspera = 0;


    printf("\nAtendimento:\n");
    Cliente *c;


    while ((c = dequeue(&f)) != NULL) {
        printf("Cliente %d | Espera: %d min\n", c->id, tempoEspera);
        tempoEspera += c->tempoAtendimento;
        free(c);
    }


    return 0;
}

#include <stdio.h>
#include <stdlib.h>


// Struct documento
typedef struct Documento {
    int id;
    int paginas;
    int prioridade;
    struct Documento *prox;
} Documento;


// Fila
typedef struct {
    Documento *inicio;
} Fila;


// Inicializar
void init(Fila *f) {
    f->inicio = NULL;
}


// Inserção ordenada por prioridade
void enqueue(Fila *f, int id, int paginas, int prioridade) {
    Documento novo = (Documento) malloc(sizeof(Documento));
    novo->id = id;
    novo->paginas = paginas;
    novo->prioridade = prioridade;
    novo->prox = NULL;


    // Inserir no início
    if (f->inicio == NULL || prioridade < f->inicio->prioridade) {
        novo->prox = f->inicio;
        f->inicio = novo;
    } else {
        Documento *atual = f->inicio;


        // percorre mantendo ordem e respeitando chegada
        while (atual->prox != NULL &&
               atual->prox->prioridade <= prioridade) {
            atual = atual->prox;
        }


        novo->prox = atual->prox;
        atual->prox = novo;
    }
}


// Dequeue
Documento* dequeue(Fila *f) {
    if (f->inicio == NULL) return NULL;


    Documento *temp = f->inicio;
    f->inicio = temp->prox;
    return temp;
}


int main() {
    Fila f;
    init(&f);


    int n;
    printf("Quantidade de documentos: ");
    scanf("%d", &n);


    for (int i = 0; i < n; i++) {
        int id, paginas, prioridade;
        printf("ID, paginas, prioridade: ");
        scanf("%d %d %d", &id, &paginas, &prioridade);
        enqueue(&f, id, paginas, prioridade);
    }


    printf("\nOrdem de impressao:\n");


    Documento *d;
    while ((d = dequeue(&f)) != NULL) {
        printf("Doc %d | Paginas: %d | Prioridade: %d\n",
               d->id, d->paginas, d->prioridade);
        free(d);
    }


    return 0;
}
